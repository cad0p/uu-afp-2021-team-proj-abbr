    9 module LibCore.Parser where
   10 
   11 import           Data.Char          (isAlphaNum, isPunctuation, isSpace)
   12 import           Data.Functor
   13 import           LibCore.Models     (Keyword (Keyword), Token (DoMap, NoToken))
   14 import           Text.Parsec
   15     ( ParseError
   16     , alphaNum
   17     , choice
   18     , many
   19     , many1
   20     , parse
   21     , satisfy
   22     , spaces
   23     , string
   24     , try
   25     )
   26 import           Text.Parsec.String (Parser)
   27 
   28 type ParseStructure = [Token]
   29 
   30 -- | The abbreviation symbol. Can later be made configurable
   31 abbSymbol :: String
   32 abbSymbol = "@@"
   33 
   34 -- | The plural symbol. String between the abbSymbol and the pluralSymbol are interpreted as plurals
   35 pluralSymbol :: String
   36 pluralSymbol = "'s"
   37 
   38 -- | Consume whitespace characters
   39 whitespace :: Parser ()
   40 whitespace = void spaces
   41 
   42 -- | Lexeme parsers consume the whitespace after them
   43 lexeme :: Parser a -> Parser a
   44 lexeme p = p <* whitespace
   45 
   46 -- | Given a string, parse it. This function can throw an error if parsing fails
   47 doParse :: String -> ParseStructure
   48 doParse s = case parseInput s of
   49   Left err -> error $ show err
   50   Right ps -> ps
   51 
   52 -- | Map a string to a list of Tokens. For example:
   53 -- >>> parseInput "@@bob"
   54 -- Right [DoMap (Keyword {keyword = "bob", plural = False})]
   55 -- >>> parseInput "@@fw's"
   56 -- Right [DoMap (Keyword {keyword = "fw", plural = True})]
   57 -- >>> parseInput "hello!"
   58 -- Right [NoToken "hello!"]
   59 parseInput :: String -> Either ParseError ParseStructure
   60 parseInput = parse (mainParser abbSymbol pluralSymbol) ""
   61 
   62 -- | The main parser tries to consume all input into a ParseStructure, given an
   63 -- | abbreviation symbol and a plural symbol
   64 mainParser :: String -> String -> Parser ParseStructure
   65 mainParser s p = do many $ choice [try $ lexeme $ pluralAbbrParser s p, lexeme $ abbrParser s, lexeme punctuationParser, lexeme noAbbrParser]
   66 
   67 -- | Inverse of the 'isSpace' function from Data.Char
   68 notSpace :: Char -> Bool
   69 notSpace c = not $ isSpace c
   70 
   71 -- | Given an abbreviation string s and a plural end string p, parse the string between it
   72 pluralAbbrParser :: String -> String -> Parser Token
   73 pluralAbbrParser s p = do
   74   void $ string s
   75   a <- many1 alphaNum
   76   void $ string p
   77   return $ DoMap $ Keyword a True
   78 
   79 -- | Given an abbreviation string s, parse the string after it
   80 abbrParser :: String -> Parser Token
   81 abbrParser s = do
   82   void $ string s
   83   a <- many1 $ satisfy isAlphaNum
   84   return $ DoMap $ Keyword a False
   85 
   86 -- | Parse any string into a token
   87 noAbbrParser :: Parser Token
   88 noAbbrParser = do
   89   a <- many1 $ satisfy notSpace
   90   return $ NoToken a
   91 
   92 -- | Parser to process punctuation. We added this because abbreviations followed by
   93 -- | punctuation without a space, such as 'hello!' were parsed as NoToken
   94 punctuationParser :: Parser Token
   95 punctuationParser = do
   96   a <- many1 $ satisfy isPunctuation
   97   return $ NoToken a
