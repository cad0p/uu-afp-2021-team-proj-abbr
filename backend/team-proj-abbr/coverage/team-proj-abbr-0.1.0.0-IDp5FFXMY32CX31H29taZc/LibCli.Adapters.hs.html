   31 -- |'KbEntry' is an entry on the file containing the Knowledge Base.
   32 data KbEntry
   33   = KbEntry
   34       { abbreviation :: String
   35       , expansion    :: String
   36       }
   37 
   38 -- TODO(tech debt): make use of the Generics as in this example:
   39 --    https://hackage.haskell.org/package/cassava-0.5.2.0/docs/Data-Csv.html#g:4
   40 -- | 'FromNamedRecord' describes how to parse the record
   41 --
   42 -- Other examples of this can be found here:
   43 -- https://github.com/haskell-hvr/cassava/blob/master/examples/
   44 instance FromNamedRecord KbEntry where
   45   parseNamedRecord r = KbEntry <$> r .: "abbreviation" <*> r .: "expansion"
   46 
   47 instance ToNamedRecord KbEntry where
   48   toNamedRecord KbEntry { abbreviation = a, expansion = e } =
   49     namedRecord ["abbreviation" .= a, "expansion" .= e]
   50 
   51 instance DefaultOrdered KbEntry where
   52   headerOrder _ = header ["abbreviation", "expansion"]
   53 
   54 
   55 {-| 'mapEntries' maps entries to a pair of ('abbreviation', 'expansion')
   56 
   57 >>> mapEntries (KbEntry { abbreviation = "abbr", expansion = "abbreviation" })
   58 (Key {keyword = "abbr", plural = False},Key {keyword = "abbreviation", plural = False})
   59 
   60 -}
   61 mapEntries :: KbEntry -> (Keyword, Keyword)
   62 mapEntries (KbEntry a e) = (pure a, pure e)
   63 
   64 -- |`mapKeywords` maps keyword pairs to their entries
   65 mapKeywordPair :: (Keyword, Keyword) -> KbEntry
   66 mapKeywordPair (Key kk _, Key vk _) = KbEntry kk vk
   67 
   68 -- |'getKnowledgeBase' parses from a CSV file to a 'KnowledgeBaseStructure'.
   69 getKnowledgeBase
   70   :: V.Vector KbEntry -- ^ it is the result of the low level parsing of the file, represented as a vector of entries
   71   -> KB.KnowledgeBaseStructure
   72 getKnowledgeBase v = KB.build $ map mapEntries $ V.toList v
