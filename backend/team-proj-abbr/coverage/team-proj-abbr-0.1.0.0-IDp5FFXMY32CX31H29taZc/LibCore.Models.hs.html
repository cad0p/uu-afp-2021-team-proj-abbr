    9 module LibCore.Models where

   11 -- | We parse a string into a NoToken if it does not match the syntax for
   12 -- | parsing. The DoMap is a string that has to be mapped. In our proposals,
   13 -- | the DoMap strings start with @@
   14 data Token
   15   = NoToken String
   16   | DoMap Keyword
   17   deriving (Eq, Ord, Show)
   18 
   19 -- | Abstract and parameterised Keyword datatype.
   20 -- Implements Functor and Applicative typeclasses.
   21 --
   22 -- Examples:
   23 --
   24 -- >>> pure "hello" :: AKeyword String
   25 -- Key {keyword = "hello", plural = False}
   26 --
   27 -- >>> (map Data.Char.toUpper) <$> (pure "hello" :: AKeyword String)
   28 -- Key {keyword = "HELLO", plural = False}
   29 --
   30 -- >>> (pure (map Data.Char.toLower)) <*> (pure "HELLO" :: AKeyword String)
   31 -- Key {keyword = "hello", plural = False}
   32 --
   33 data AKeyword a
   34   = Key
   35       { keyword :: a
   36       , plural  :: Bool
   37       }
   38   deriving (Eq, Ord, Show)
   39 
   40 instance Functor AKeyword where
   41   fmap f (Key k p) = Key (f k) p
   42 
   43 instance Applicative AKeyword where
   44   pure a = Key a False
   45   (Key f p) <*> (Key a p') = Key (f a) (p && p') -- joining both plurals.
   46 
   47 -- | A Keyword is a Token that can be mapped. A Keyword can be plural
   48 type Keyword = AKeyword String
   49 
   50 -- | Reserved domain-level error indications.
   51 newtype Error
   52   = StandardError String
   53   deriving (Show)
