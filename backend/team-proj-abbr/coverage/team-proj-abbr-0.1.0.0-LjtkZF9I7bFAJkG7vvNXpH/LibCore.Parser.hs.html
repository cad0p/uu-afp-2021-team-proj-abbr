   26 -- | The abbreviation symbol. Can later be made configurable
   27 abbSymbol :: String

   30 -- | The plural symbol. String between the abbSymbol and the pluralSymbol are interpreted as plurals

   34 -- | Consume whitespace characters
   35 whitespace :: Parser ()
   36 whitespace = void spaces
   37 
   38 -- | Lexeme parsers consume the whitespace after them
   39 lexeme :: Parser a -> Parser a
   40 lexeme p = p <* whitespace

   42 -- | Given a string, parse it. This function can throw an error if parsing fails
   43 doParse :: String -> ParseStructure
   44 doParse s = case parseInput s of
   45   Left err -> error $ show err
   46   Right ps -> ps
   47 
   48 -- | Map a string to a list of Tokens. For example:
   49 -- >>> parseInput "@@bob"
   50 -- Right [DoMap (Keyword "bob")]
   51 -- >>> parseInput "@@fw's"
   52 -- Right [DoMap (Plural "fw")]
   53 -- >>> parseInput "hello!"
   54 -- Right [NoToken "hello"]
   55 parseInput :: String -> Either ParseError ParseStructure
   56 parseInput = parse (mainParser abbSymbol pluralSymbol) ""
   57 
   58 -- | The main parser tries to consume all input into a ParseStructure, given an
   59 -- | abbreviation symbol and a plural symbol
   60 mainParser :: String -> String -> Parser ParseStructure
   61 mainParser s p = do many $ choice [try $lexeme $ pluralAbbrParser s p, lexeme $ abbrParser s, lexeme noAbbrParser]
   62 
   63 -- | Inverse of the 'isSpace' function from Data.Char
   64 notSpace :: Char -> Bool
   65 notSpace c = not $ isSpace c
   66 
   67 -- | Given an abbreviation string s and a plural end string p, parse the string between it
   68 pluralAbbrParser :: String -> String -> Parser Token
   69 pluralAbbrParser s p = do
   70   void $ string s
   71   a <- many1 alphaNum
   72   void $ string p
   73   return $ DoMap $ Plural a
   74 
   75 -- | Given an abbreviation string s, parse the string after it
   76 abbrParser :: String -> Parser Token
   77 abbrParser s = do
   78   void $ string s
   79   a <- many1 $ satisfy notSpace
   80   return $ DoMap $ Keyword a
   81 
   82 -- | Parse any string into a token
   83 noAbbrParser :: Parser Token
   84 noAbbrParser = do
   85   a <- many1 $ satisfy notSpace
   86   return $ NoToken a
