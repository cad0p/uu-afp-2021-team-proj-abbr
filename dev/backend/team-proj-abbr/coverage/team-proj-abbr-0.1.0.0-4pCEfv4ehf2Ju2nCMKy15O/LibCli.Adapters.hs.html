   26 -- |'KbEntry' is an entry on the file containing the Knowledge Base.
   27 data KbEntry
   28   = KbEntry
   29       { abbreviation :: String
   30       , expansion    :: String
   31       }
   32 
   33 -- | 'FromNamedRecord' describes how to parse the record
   34 --
   35 -- Other examples of this can be found here:
   36 -- https://github.com/haskell-hvr/cassava/blob/master/examples/
   37 instance FromNamedRecord KbEntry where
   38   parseNamedRecord r = KbEntry <$> r .: "abbreviation" <*> r .: "expansion"
   39 
   40 
   41 {-| 'mapEntries' maps entries to a pair of ('abbreviation', 'expansion')
   42 
   43 >>> mapEntries (KbEntry { abbreviation = "abbr", expansion = "abbreviation" })
   44 (Keyword {keyword = "abbr", plural = False},Keyword {keyword = "abbreviation", plural = False})
   45 
   46 -}
   47 mapEntries :: KbEntry -> (Keyword, Keyword)
   48 mapEntries e =
   49   ( (Keyword { keyword = abbreviation e, plural = False })
   50   , (Keyword { keyword = expansion e, plural = False })
   51   )
   52 
   53 -- |'getKnowledgeBase' parses from a CSV file to a 'KnowledgeBaseStructure'.
   54 getKnowledgeBase
   55   :: V.Vector KbEntry -- ^ it is the result of the low level parsing of the file, represented as a vector of entries
   56   -> KB.KnowledgeBaseStructure
   57 getKnowledgeBase v = KB.build $ map mapEntries $ V.toList v
