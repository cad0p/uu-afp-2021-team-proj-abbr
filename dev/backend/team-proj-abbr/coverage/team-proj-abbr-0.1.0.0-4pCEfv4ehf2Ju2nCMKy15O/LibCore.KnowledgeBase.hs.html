   13 module LibCore.KnowledgeBase where

   22 type KnowledgeBaseStructure = Map Keyword Keyword
   23 
   24 -- | 'KbEntry' is an entry on the file containing the KB
   25 data KbEntry
   26   = KbEntry
   27       { abbreviation :: String
   28       , expansion    :: String
   29       }
   30 
   31 -- | 'FromNamedRecord' describes how to parse the record
   32 --
   33 -- Other examples of this can be found here:
   34 -- https://github.com/haskell-hvr/cassava/blob/master/examples/
   35 instance FromNamedRecord KbEntry where
   36   parseNamedRecord r = KbEntry <$> r .: "abbreviation" <*> r .: "expansion"
   37 
   38 
   39 {-| 'mapEntries' maps entries to a pair of ('abbreviation', 'expansion')
   40 
   41 >>> mapEntries (KbEntry { abbreviation = "abbr", expansion = "abbreviation" })
   42 (Keyword {keyword = "abbr", plural = False},Keyword {keyword = "abbreviation", plural = False})
   43 
   44 -}
   45 mapEntries :: KbEntry -> (Keyword, Keyword)
   46 mapEntries e =
   47   ( (Keyword { keyword = abbreviation e, plural = False })
   48   , (Keyword { keyword = expansion e, plural = False })
   49   )
   50 
   51

   53 
   54 {-| 'getKnowledgeBase' parses from a CSV file to a 'KnowledgeBaseStructure'
   55 
   56 -}
   57 getKnowledgeBase
   58   :: V.Vector KbEntry -- ^ it is the result of the low level parsing of the file, represented as a vector of entries
   59   -> KnowledgeBaseStructure
   60 getKnowledgeBase v = fromList (map mapEntries (V.toList v))
