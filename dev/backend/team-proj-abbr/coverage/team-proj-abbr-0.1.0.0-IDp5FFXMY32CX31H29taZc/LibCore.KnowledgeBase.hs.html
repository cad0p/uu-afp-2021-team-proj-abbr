   27 -- | Build a new Knowledge Base from the supplied keyword pair list.
   28 build :: [(Keyword, Keyword)] -> KnowledgeBaseStructure
   29 build = M.fromList

   33 -- |Get the list of all the stored records in the Knowledge Base.
   34 --
   35 -- Examples:
   36 --
   37 -- >>> testKB = build [(Keyword {keyword = "brb", plural = False}, Keyword {keyword = "be right back", plural = False})]
   38 -- >>> listAll testKB
   39 -- [(Keyword {keyword = "brb", plural = False},Keyword {keyword = "be right back", plural = False})]
   40 listAll :: KnowledgeBaseStructure -> [(Keyword, Keyword)]
   41 listAll = M.toList
   42

   44 -- |Retrieve a single element by its keyword.
   45 --
   46 -- Examples
   47 --
   48 -- >>> testKB = build [(Keyword {keyword = "brb", plural = False}, Keyword {keyword = "be right back", plural = False})]
   49 -- >>> get testKB (Keyword {keyword = "brb", plural = False})
   50 -- Right (Keyword {keyword = "be right back", plural = False})
   51 --
   52 -- >>> get testKB (Keyword {keyword = "beb", plural = False})
   53 -- Left (StandardError "no record found for this keyword : Keyword {keyword = \"beb\", plural = False}")
   54 get :: KnowledgeBaseStructure -> Keyword -> Either Error Keyword
   55 get kb k = case M.lookup k kb of
   56   Nothing ->
   57     Left $ StandardError $ "no record found for this keyword : " ++ show k
   58   Just x -> Right x
   59

   61 -- |Add new item to the Knowledge Base.
   62 -- Adds the item to the KB as a new record if the key was notseen before,
   63 -- and replaces the existing record otherwise.
   64 --
   65 -- Examples
   66 --
   67 -- >>> testKB = build [(Keyword {keyword = "brb", plural = False}, Keyword {keyword = "be right back", plural = False})]
   68 -- >>> add testKB (Keyword {keyword = "btw", plural = False}) (Keyword {keyword = "by the way", plural = False})
   69 -- (Keyword {keyword = "by the way", plural = False},fromList [(Keyword {keyword = "brb", plural = False},Keyword {keyword = "be right back", plural = False}),(Keyword {keyword = "btw", plural = False},Keyword {keyword = "by the way", plural = False})])
   70 --
   71 -- >>> add testKB (Keyword {keyword = "btw", plural = False}) (Keyword {keyword = "by the way!", plural = False})
   72 -- (Keyword {keyword = "by the way!", plural = False},fromList [(Keyword {keyword = "brb", plural = False},Keyword {keyword = "be right back", plural = False}),(Keyword {keyword = "btw", plural = False},Keyword {keyword = "by the way!", plural = False})])
   73 add
   74   :: KnowledgeBaseStructure
   75   -> Keyword
   76   -> Keyword
   77   -> (Keyword, KnowledgeBaseStructure)
   78 add kb k r = (r, M.insert k r kb)
   79 
   80 
   81 -- |Updates the record at given key with the new value.
   82 -- Only full updates are supported.
   83 -- If the key is not in the KB, returns an error.
   84 --
   85 -- Examples:
   86 --
   87 -- >>> testKB = build [(Keyword {keyword = "brb", plural = False}, Keyword {keyword = "be right back", plural = False})]
   88 -- >>> put testKB (Keyword {keyword = "brb", plural = False}) (Keyword {keyword = "BE RIGHT BACK", plural = False})
   89 -- Right (Keyword {keyword = "BE RIGHT BACK", plural = False},fromList [(Keyword {keyword = "brb", plural = False},Keyword {keyword = "BE RIGHT BACK", plural = False})])
   90 --
   91 -- >>> put testKB (Keyword {keyword = "btw", plural = False}) (Keyword {keyword = "by the way!", plural = False})
   92 -- Left (StandardError "no record found for this keyword : Keyword {keyword = \"btw\", plural = False}")
   93 put
   94   :: KnowledgeBaseStructure
   95   -> Keyword
   96   -> Keyword
   97   -> Either Error (Keyword, KnowledgeBaseStructure)
   98 put kb k r = case get kb k of
   99   Left  er -> Left er
  100   Right _  -> Right (r, M.update (\_ -> Just r) k kb)

  103 -- |Removes an existing item at the given key form the Knowledge Base.
  104 -- If the key is not present, return as error.
  105 --
  106 -- Examples:
  107 --
  108 -- >>> testKB = build [(Keyword {keyword = "brb", plural = False}, Keyword {keyword = "be right back", plural = False})]
  109 -- >>> remove testKB (Keyword {keyword = "brb", plural = False})
  110 -- Right (fromList [])
  111 --
  112 -- >>> remove testKB (Keyword {keyword = "btw", plural = False})
  113 -- Left (StandardError "no record found for this keyword : Keyword {keyword = \"btw\", plural = False}")
  114 remove
  115   :: KnowledgeBaseStructure -> Keyword -> Either Error KnowledgeBaseStructure
  116 remove kb k = case get kb k of
  117   Left  er -> Left er
  118   Right _  -> Right $ M.delete k kb
