   26 -- | Build a new Knowledge Base from the supplied keyword pair list.
   27 build :: [(Keyword, Keyword)] -> KnowledgeBaseStructure
   28 build = M.fromList

   32 -- |Get the list of all the stored records in the Knowledge Base.
   33 --
   34 -- Examples:
   35 --
   36 -- >>> testKB = build [(pure "brb", pure "be right back")]
   37 -- >>> listAll testKB
   38 -- [(KeywordBody {keyword = "brb", plural = False},KeywordBody {keyword = "be right back", plural = False})]
   39 --
   40 listAll :: KnowledgeBaseStructure -> [(Keyword, Keyword)]
   41 listAll = M.toList
   42

   44 -- |Retrieve a single element by its keyword.
   45 --
   46 -- Examples
   47 --
   48 -- >>> testKB = build [(pure "brb", pure "be right back")]
   49 -- >>> get testKB $ pure "brb"
   50 -- Right (KeywordBody {keyword = "be right back", plural = False})
   51 --
   52 -- >>> get testKB $ pure "beb"
   53 -- Left (StandardError "no record found for this keyword : KeywordBody {keyword = \"beb\", plural = False}")
   54 --
   55 get :: KnowledgeBaseStructure -> Keyword -> Either Error Keyword
   56 get kb k = case M.lookup k kb of
   57   Nothing ->
   58     Left $ StandardError $ "no record found for this keyword : " ++ show k
   59   Just x -> Right x
   60

   62 -- |Add new item to the Knowledge Base.
   63 -- Adds the item to the KB as a new record if the key was notseen before,
   64 -- and replaces the existing record otherwise.
   65 --
   66 -- Examples
   67 --
   68 -- >>> testKB = build [(pure "brb", pure "be right back")]
   69 -- >>> add testKB (pure "btw") (pure "by the way")
   70 -- (KeywordBody {keyword = "by the way", plural = False},fromList [(KeywordBody {keyword = "brb", plural = False},KeywordBody {keyword = "be right back", plural = False}),(KeywordBody {keyword = "btw", plural = False},KeywordBody {keyword = "by the way", plural = False})])
   71 --
   72 -- >>> add testKB (pure "btw") (pure "by the way!")
   73 -- (KeywordBody {keyword = "by the way!", plural = False},fromList [(KeywordBody {keyword = "brb", plural = False},KeywordBody {keyword = "be right back", plural = False}),(KeywordBody {keyword = "btw", plural = False},KeywordBody {keyword = "by the way!", plural = False})])
   74 --
   75 add
   76   :: KnowledgeBaseStructure
   77   -> Keyword
   78   -> Keyword
   79   -> (Keyword, KnowledgeBaseStructure)
   80 add kb k r = (r, M.insert k r kb)
   81 
   82 
   83 -- |Updates the record at given key with the new value.
   84 -- Only full updates are supported.
   85 -- If the key is not in the KB, returns an error.
   86 --
   87 -- Examples:
   88 --
   89 -- >>> testKB = build [(pure "brb", pure "be right back")]
   90 -- >>> put testKB (pure "brb") (pure "BE RIGHT BACK")
   91 -- Right (KeywordBody {keyword = "BE RIGHT BACK", plural = False},fromList [(KeywordBody {keyword = "brb", plural = False},KeywordBody {keyword = "BE RIGHT BACK", plural = False})])
   92 --
   93 -- >>> put testKB (pure "btw") (pure "by the way!")
   94 -- Left (StandardError "no record found for this keyword : KeywordBody {keyword = \"btw\", plural = False}")
   95 --
   96 put
   97   :: KnowledgeBaseStructure
   98   -> Keyword
   99   -> Keyword
  100   -> Either Error (Keyword, KnowledgeBaseStructure)
  101 put kb k r = case get kb k of
  102   Left  er -> Left er
  103   Right _  -> Right (r, M.update (\_ -> Just r) k kb)
  104 
  105 
  106 -- |Removes an existing item at the given key form the Knowledge Base.
  107 -- If the key is not present, return as error.
  108 --
  109 -- Examples:
  110 --
  111 -- >>> testKB = build [(pure "brb", pure "be right back")]
  112 -- >>> remove testKB $ pure "brb"
  113 -- Right (fromList [])
  114 --
  115 -- >>> remove testKB $ pure "btw"
  116 -- Left (StandardError "no record found for this keyword : KeywordBody {keyword = \"btw\", plural = False}")
  117 --
  118 remove
  119   :: KnowledgeBaseStructure -> Keyword -> Either Error KnowledgeBaseStructure
  120 remove kb k = case get kb k of
  121   Left  er -> Left er
  122   Right _  -> Right $ M.delete k kb
